class Solution {
    public static int[] ll = {0, 3};
    public static int[] rl = {2, 3};
    
    public String solution(int[] numbers, String hand) {
        String answer = "";
        int h = hand.equals("right")?0:1;
        for(int i : numbers){
            String tmp = "";
            if(i==1){
                tmp = "L";
                ll[0] = 0;
                ll[1] = 0;
            }else if(i==4){
                tmp = "L";
                ll[0] = 0;
                ll[1] = 1;
            }else if(i==7){
                tmp = "L";
                ll[0] = 0;
                ll[1] = 2;
            }else if(i==3){
                tmp = "R";
                rl[0] = 2;
                rl[1] = 0;
            }else if(i==6){
                tmp = "R";
                rl[0] = 2;
                rl[1] = 1;
            }else if(i==9){
                tmp = "R";
                rl[0] = 2;
                rl[1] = 2;
            }else if(i==2){
                int ld = Math.abs(ll[0]-1)+Math.abs(ll[1]-0);
                int rd = Math.abs(rl[0]-1)+Math.abs(rl[1]-0);
                if(ld>rd){
                    tmp = "R";
                    rl[0] = 1;
                    rl[1] = 0;
                }else if(ld<rd){
                    tmp = "L";
                    ll[0] = 1;
                    ll[1] = 0;
                }else{
                    if(h == 0){
                        tmp = "R";
                        rl[0] = 1;
                        rl[1] = 0;
                    }else{
                        tmp = "L";
                        ll[0] = 1;
                        ll[1] = 0;
                    }
                }
            }else if(i==5){
                int ld = Math.abs(ll[0]-1)+Math.abs(ll[1]-1);
                int rd = Math.abs(rl[0]-1)+Math.abs(rl[1]-1);
                if(ld>rd){
                    tmp = "R";
                    rl[0] = 1;
                    rl[1] = 1;
                }else if(ld<rd){
                    tmp = "L";
                    ll[0] = 1;
                    ll[1] = 1;
                }else{
                    if(h == 0){
                        tmp = "R";
                        rl[0] = 1;
                        rl[1] = 1;
                    }else{
                        tmp = "L";
                        ll[0] = 1;
                        ll[1] = 1;
                    }
                }
            }else if(i==8){
                int ld = Math.abs(ll[0]-1)+Math.abs(ll[1]-2);
                int rd = Math.abs(rl[0]-1)+Math.abs(rl[1]-2);
                if(ld>rd){
                    tmp = "R";
                    rl[0] = 1;
                    rl[1] = 2;
                }else if(ld<rd){
                    tmp = "L";
                    ll[0] = 1;
                    ll[1] = 2;
                }else{
                    if(h == 0){
                        tmp = "R";
                        rl[0] = 1;
                        rl[1] = 2;
                    }else{
                        tmp = "L";
                        ll[0] = 1;
                        ll[1] = 2;
                    }
                }
            }else if(i==0){
                int ld = Math.abs(ll[0]-1)+Math.abs(ll[1]-3);
                int rd = Math.abs(rl[0]-1)+Math.abs(rl[1]-3);
                if(ld>rd){
                    tmp = "R";
                    rl[0] = 1;
                    rl[1] = 3;
                }else if(ld<rd){
                    tmp = "L";
                    ll[0] = 1;
                    ll[1] = 3;
                }else{
                    if(h == 0){
                        tmp = "R";
                        rl[0] = 1;
                        rl[1] = 3;
                    }else{
                        tmp = "L";
                        ll[0] = 1;
                        ll[1] = 3;
                    }
                }
            }
            answer += tmp;
        }
        return answer;
    }
}
